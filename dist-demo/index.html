Installation & Usage

npm install s-console
import { sconsole } from 's-console';

const console = new sconsole('container-id', {
    theme: 'dark',
    fontSize: '16px',
    fontFamily: 'monospace'
});

console.addCommand('hello', () => {
    console.appendToConsole('Hello World!');
});
